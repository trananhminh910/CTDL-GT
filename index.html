#include<
stdio.h
>
struct SinhVien{
	char maSV[15], tenSV[20], lop[15];
	float dtb;
};
struct Node_BnTr{
	SinhVien Data;
	Node_BnTr *Node_TrL,*Node_TrR;
};
Node_BnTr *Add_Node(Node_BnTr *Tr, Node_BnTr *node)
{
	Node_BnTr *TMP,*tmp;
	if(Tr==NULL)
		Tr=node;
	else
	{
		tmp=Tr;
		while(tmp != NULL)
		{
			TMP = tmp ;
			if(tmp->Data.dtb>node->Data.dtb)
				tmp=tmp->Node_TrL;
			else
				tmp=tmp->Node_TrR;
		}
		if(TMP->Data.dtb>node->Data.dtb)
			TMP->Node_TrL=node;
		else
			TMP->Node_TrR=node;
	}
	return Tr;
}

Node_BnTr *Input_Add(Node_BnTr *Tr)
{
	Node_BnTr *p;
	int choose;
	int i=1;
	do
	{
	p= new Node_BnTr;
	printf("\n Sinh Vien %d\n ",i);
	printf("Nhap ma SV: ");
	scanf("%s",&p->Data.maSV);
	printf("Nhap ten sv: ");
	fflush(stdin);
	gets(p->Data.tenSV);
	printf("Nhap lop sv: ");
	scanf("%s",&p->Data.lop);
	printf("Nhap dtb: ");
	scanf("%f",&p->Data.dtb);
	p->Node_TrL=NULL;
	p->Node_TrR=NULL;
	Tr=Add_Node(Tr,p);
	i++;
	printf("1.Nhap SV tiep theo \n");
	printf("2.Quay lai Menu \n");
	printf("Nhap so: ");
	scanf("%d",&choose);
	}
	while(choose == 1);
	return Tr;
}
Node_BnTr *Input(Node_BnTr *Tr)
{
	Tr=NULL;
	Tr=Input_Add(Tr);
	return Tr;
}
void Output(Node_BnTr *Tr)
{
	if(Tr!=NULL)
	{
		Output(Tr->Node_TrR);
		Output(Tr->Node_TrL);
		printf(" %15s | %20s | %15s | %2.2f \n",Tr->Data.maSV,Tr->Data.tenSV,Tr->Data.lop,Tr->Data.dtb);	
		
	}
}
int Count(Node_BnTr *Tr)
{
	if(Tr==NULL)
		return 0;
	if(Tr->Data.dtb>5.5)
		return Count(Tr->Node_TrL)+Count(Tr->Node_TrR)+1;
	else
		return Count(Tr->Node_TrL)+Count(Tr->Node_TrR);
}
void Output_Condition(Node_BnTr *Tr)
{
	if(Tr!=NULL)
	{
		Output_Condition(Tr->Node_TrR);
		Output_Condition(Tr->Node_TrL);
		if(Tr->Data.dtb>5.5)
			printf(" %15s | %20s | %15s | %2.2f \n",Tr->Data.maSV,Tr->Data.tenSV,Tr->Data.lop,Tr->Data.dtb);	
		
	}
}

Node_BnTr *Search(Node_BnTr *Tr)
{
	Node_BnTr *p;
	if(Tr==NULL)
		return NULL;
	else
	{
		p=Tr;
		while(p!=NULL && p->Data.dtb>5.5)
		{
			if(p->Data.dtb>5.5)
				p=p->Node_TrL;
			else
				p=p->Node_TrR;
		}
	if(p!=NULL)
		return p;
	else
		return NULL;
	}
}
Node_BnTr *Delete_Node_Root(Node_BnTr *Tr,Node_BnTr *tmp)
{
	Node_BnTr *R;
	if(tmp->Node_TrL!=NULL)
	{
		R=Tr->Node_TrR;
		Tr= Tr->Node_TrL;
		tmp->Node_TrL=NULL;
		tmp->Node_TrR=NULL;
		tmp=Tr;
		while(tmp->Node_TrR!=NULL)
			tmp=tmp->Node_TrR;
		tmp->Node_TrR=R;
	}
	else
	{
		Tr=Tr->Node_TrR;
		tmp->Node_TrR=NULL;
	}
	return Tr;
}
Node_BnTr *Delete_Node_Leaves(Node_BnTr *Tr,Node_BnTr *tmp)
{
	Node_BnTr *t;
	t=Tr;
	while(t->Node_TrR!=tmp && t->Node_TrL!=tmp)
	{
		if(t->Data.dtb>tmp->Data.dtb)
			t=t->Node_TrL;
		else
			t=t->Node_TrR;
	}
	if(t->Node_TrR==tmp)
		t->Node_TrR=NULL;
	else
		t->Node_TrL=NULL;
	return Tr;
}
Node_BnTr *Delete_Node_Right(Node_BnTr *Tr,Node_BnTr *tmp)
{
	Node_BnTr *R,*t;
	t=Tr;
	while(t->Node_TrR!=tmp && t->Node_TrR!=tmp)
		if(t->Data.dtb>tmp->Data.dtb)
			t=t->Node_TrL;
		else
			t=t->Node_TrR;
	R=tmp->Node_TrR;
	if(t->Node_TrL==tmp)
		t->Node_TrL=R;
	else
		t->Node_TrR=R;
	tmp->Node_TrR=NULL;
	tmp->Node_TrL=NULL;
	return Tr;
}
Node_BnTr *Delete_Node_Left(Node_BnTr *Tr,Node_BnTr *tmp)
{
	Node_BnTr *L,*t;
	t=Tr;	
	while(t->Node_TrR!=tmp && t->Node_TrL!=tmp)
		if(t->Data.dtb>tmp->Data.dtb)
			t=t->Node_TrL;
		else
			t=t->Node_TrR;
	L=tmp->Node_TrL;
	if(t->Node_TrL==tmp)
		t->Node_TrL=L;
	else
		t->Node_TrR=L;
	tmp->Node_TrL=NULL;
	tmp->Node_TrR=NULL;
	return Tr;
}
Node_BnTr *Delete_Node_RL(Node_BnTr *Tr,Node_BnTr *tmp)
{
	Node_BnTr *t;
	t=Tr;
	while(t->Node_TrR!=tmp && t->Node_TrL!=tmp)
		if(t->Data.dtb>tmp->Data.dtb)
			t=t->Node_TrL;
		else
			t=t->Node_TrR;
	if(t->Node_TrL==tmp)
	{
		t->Node_TrL=tmp->Node_TrL;
		t=t->Node_TrL;
	}
	else
	{
		t->Node_TrR=tmp->Node_TrL;
		t=t->Node_TrR;
	}
	while(t->Node_TrR!=NULL)
		t=t->Node_TrR;
	t->Node_TrR=tmp->Node_TrR;
	tmp->Node_TrR=NULL;
	tmp->Node_TrL=NULL;
	return Tr;
}
Node_BnTr *Delete_Node(Node_BnTr *Tr)
{
	Node_BnTr *tmp;
	while(Search(Tr)!=NULL)
	{
		tmp=Search(Tr);
		if(tmp==Tr)
			Tr=Delete_Node_Root(Tr,tmp);
		else
		{
			if(tmp->Node_TrL==NULL && tmp->Node_TrR==NULL)
				Tr=Delete_Node_Leaves(Tr,tmp);
			else
				if(tmp->Node_TrL!=NULL && tmp->Node_TrR!=NULL)
					Tr=Delete_Node_RL(Tr,tmp);
				else
					if(tmp->Node_TrL!=NULL)
						Tr=Delete_Node_Right(Tr,tmp);
					else
						Tr=Delete_Node_Left(Tr,tmp);
		}
		
	}
	return Tr;
}

NHẬN XÉT & RÚT KINH NGHIỆM

Nhận xét:

- Ngày đầu tiên các thành viên họp lại và phân chia công việc cho mỗi người

-Sau khi hoàn thành xong, Họp nhóm để tập trung các hàm lại hoàn thiện bài tập nhóm

-Trong quá trình thực hiện nhóm hơi gặp vấn đề trong hàm xóa. Nhóm đã cùng nhau giải quyết vấn đề

-Tuy đã làm thành một bài hoàn chỉnh nhưng chưa thực hiện đúng yêu cầu làm Menu nên nhóm phải tập trung hoàn thiện

Rút kinh nghiệm:

-Cần chuẩn bị kỹ công việc mình được phân công trước lúc đi học nhóm

-Nên lắng nghe kỹ yêu cầu bài tập để tối ưu hóa thời gian
